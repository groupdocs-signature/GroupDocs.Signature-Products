                string id = "ff988ab1-7403-4c8d-8db7-f2a56b9f8530";

                ImageSignature signatureToUpdate = new ImageSignature()
                {
                    // set up particular signature id

                    // specify signature width

                    // specify signature height
                    Height = 250,
                    // set left position

                    Top = 10
                };
