          ```cs
          using (Signature signature = new Signature("D:\\sample.pdf"))

          TextSignOptions options = new TextSignOptions("John Smith")
          {
          // Metin rengini ayarla
          ForeColor = Color.Red
          };
          // belgeyi dosyaya imzala
          signature.Sign("D:\\signed.pdf", options);
          }
          ```
